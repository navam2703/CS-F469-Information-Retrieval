class InvertedIndex:
    """ InvertedIndex data structure class with insert, search and find functions"""
    def __init__(self):
        """
        Initializes the inverted index data structure.
        index: A dictionary that maps words to a list of doc_ids
        voc_list: A list of all the unique words in the index
        voc_count: The number of words in the index
        """
        self.index = {}
        self.voc_list = []
        self.voc_count = 0
    
    def insert(self,word,doc_id):
        """
        Inserts a word into the inverted index and adds a doc_id to the list 
        \nArgs: word, doc_id
        """
        if word not in self.index:
            self.index[word] = [doc_id]
            if word[-1] == "$":
              self.voc_list.append(word)
            self.voc_count += 1
        if doc_id not in self.index[word]:
          self.index[word].append(doc_id)
    def search(self,word):
      """
      Searches the inverted index for a word and returns the list of doc_ids
      \n used in case of wildcard queries
      \n Args: word
      """
      final_output = []
      for w in self.index.keys():
        if w.startswith(word):
          final_output.append({ get_main(w): self.index[w]})
      return final_output
    def find(self,word):
        """
        Find if a word exists in the inverted index
        and returns the list of doc_ids otherwise None
        \n used in case of regular queries 
        \n Args: word
        """
        if word in self.index:
            return self.index[word]
        else:
            return None
    def getSize(self):
      """
      Returns the number of words in the inverted index
      """
      return len(self.index)

def get_main(word):
  """get the original word where $ is the last character,rotate word till we reach state 
    \n Args : word
  """

  while(word[-1] != '$'):
    word = rotateString(word,1)
  return word



def convert(a,doc_id):
  """ 
  Another approach to just use ints instead of a full list to store doc ids in a binary form
  for example if we store 9 as the value, the binary value will be 
  00001001 which means the 1st and 4th docs have the word
  """

  l = 1<<doc_id
  return a|l

def edit_distance(s1, s2):
    """ Edit distance function using dynamic programming 
    \nCan perform 3 operations: insert, delete, replace
    \n Args: s1(word 1), s2 (word 2)
    """
    m = len(s1)
    n = len(s2)
    dp = np.zeros((m+1, n+1))
    for i in range(m+1):
        dp[i][0] = i
    for j in range(n+1):
        dp[0][j] = j
    for i in range(1, m+1):
        for j in range(1, n+1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = 1 + min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])
    return dp[m][n]

def kClosest(q,index,k = 5):
    """
    Return k (default 5) words in the corpus with least edit distance to the input query 
    runs on against all the query terms where difference in length is 2
    \n Args: query, index, k (default 5)
    """
    output = {}
    q = q + "$"
    l = len(q)
    for w in index.voc_list:
      if abs(l - len(w)) <= 2:
        d = edit_distance(q,w)
        output[w] = d
    s = sorted(output.items(), key = lambda kv : kv[1])
    return s[:k]


def getCorrect(q_t,index):
  """
  function to get the correct word for each query term,
  if query term exists in index return q_t directly,
  otherwise find k closest values and return the first one with the smallest distance
  """
  if index.find(q_t):
    return q_t
  q_ed = kClosest(q_t,index)
  for i in q_ed:
    if int(i[1]) == 0:
      return q_t
    elif int(i[1]) <= 5:
      return i[0]
    else:
      return q_t

def and_op(l1,l2):
  """
  perform the and intersection between the two lists l1 and l2
  \nArgs: l1 posting list for term 1 , l2 posting list for term 2 
  """
  i = 0
  j = 0
  intersection = []
  while i < len(l1) and j < len(l2):
    if(l1[i] == l2[j]):
      intersection.append(l1[i])
      i += 1
      j += 1
    elif (l1[i] < l2[j]):
      i += 1
    else:
      j += 1
  return intersection

def or_op(l1,l2):
  """
  perform the or operation on the two lists
  l2 or l1
  \nArgs: l1 posting list for term 1 , l2 posting list for term 2
  """
  for i in l2:
    if i not in l1:
      l1.append(i)
  return l1

def not_op(l,size):
  """
  negate the list and create a new list with all elements in space size 
  and not in l
  \nArgs: l posting list , size (no of documents in corpus)
  """
  l_not = []
  for i in range(1,size+1):
    if i not in l:
      l_not.append(i)
  return l_not

class InvertedIndex:
    """ InvertedIndex data structure class with insert, search and find functions"""
    def __init__(self):
        """
        Initializes the inverted index data structure.
        index: A dictionary that maps words to a list of doc_ids
        voc_list: A list of all the unique words in the index
        voc_count: The number of words in the index
        """
        self.index = {}
        self.voc_list = []
        self.voc_count = 0
    
    def insert(self,word,doc_id):
        """
        Inserts a word into the inverted index and adds a doc_id to the list 
        \nArgs: word, doc_id
        """
        if word not in self.index:
            self.index[word] = [doc_id]
            if word[-1] == "$":
              self.voc_list.append(word)
            self.voc_count += 1
        if doc_id not in self.index[word]:
          self.index[word].append(doc_id)
    def search(self,word):
      """
      Searches the inverted index for a word and returns the list of doc_ids
      \n used in case of wildcard queries
      \n Args: word
      """
      final_output = []
      for w in self.index.keys():
        if w.startswith(word):
          final_output.append({ get_main(w): self.index[w]})
      return final_output
    def find(self,word):
        """
